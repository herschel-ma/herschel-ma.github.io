2019-07-21

selenium + chromedriver

  • 使用selenium可以很好的实现模拟登录等，本次练习使用他
  • 由于时间原因比较粗糙
  • 由于在win7系统下进行了爬取，所以使用chromedriver时需要指定executable_path,翻译过来就是可执行路径，之前已经吧chromedriver.exe可执行文件放在了pyhon36\Scripts目录下（该目录是我win7系统下的Python安装路径，并在PATH中添加了环境变量）

  • oop

    基本思路

  • 初始页面中，利用selenium找到职位详情页的tag,然后解析找到detail_url

  • 进入详情页解析（值得注意的是在进入前，driver要打开一个新的窗口，解析后，要关闭这个窗口，然后重复之前的动作）

  • 一页搞完再到另一页

  • 比较蛋疼的是数据清洗，不过还好

  • 其实selenium的主要用途在这里没有很好的体现出来，他的强大实在模拟登录

from selenium import webdriver
from lxml import etree
import time
import re
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC
from selenium.webdriver.common.by import By

class Lagou(object):
    driver_path = "D:\Downloads\chromedriver\chromedriver.exe"
    def __init__(self):
        self.driver = webdriver.Chrome(
            executable_path=Lagou.driver_path)
        self.start_url = "https://www.lagou.com/zhaopin/Python/?labelWords=label"
        self.info = {}


    def run(self):
        self.driver.get(self.start_url)
        while True:
            source = self.driver.page_source
            self.parse_list_page(source)
            # driver.find_element_by_xpath("//ul[@class='clearfix']//a[@class='tab']")
            next_btn = self.driver.find_element_by_xpath("//div[@class='pager_container']/a[last()]")
            if "pager_next_disabled" in next_btn.get_attribute("class"):
                break
            else:
                next_btn.click()
            time.sleep(2)

    def parse_list_page(self,source):
        urls = etree.HTML(source).xpath("//div[@class='list_item_top']//a[@class='position_link']/@href")
        for detail_url in urls:
            self.request_detail_page(detail_url)
            time.sleep(2)


    def request_detail_page(self, url):
        self.driver.execute_script("window.open('%s')" %url)
        self.driver.switch_to.window(self.driver.window_handles[1])
        WebDriverWait(self.driver, timeout=10).until(
            EC.presence_of_element_located((By.XPATH, "//div[@class='job-name']//h2[@class='name']"))
        )
        source = self.driver.page_source
        self.parse_detail_page(source)
        # 关闭当前这个详情页
        self.driver.close()
        # 继续切换回职位列表
        self.driver.switch_to.window(self.driver.window_handles[0])


    def parse_detail_page(self, source):
        html = etree.HTML(source)
        position = html.xpath("//div[@class='job-name']//h2[@class='name']/text()")[0]
        youhuo = html.xpath("//dd[@class='job-advantage']//text()")
        # youhuo = re.sub(r"[\s]","",str(youhuo))
        youhuo = [x.strip() for x in youhuo if x.strip() != '']
        describe = html.xpath("//dd[@class='job_bt']//div[@class='job-detail']/p//text()")
        describe = [''.join(x.split()) for x in describe]
        location = html.xpath("//div[@class='work_addr']//text()")
        location = [''.join(x.split()) for x in location]
        company = html.xpath("//div[@class='job_company_content']/h3/em/text()")
        company = [x.strip() for x in company if x.strip() != '']

        self.info["positon"] = position
        self.info["youhuo"] = youhuo
        self.info["describe"] = describe
        self.info["location"] = location
        self.info["company"] = company
        print("--"*50)
        print(self.info)



if __name__ == "__main__":
    lagou = Lagou()
    lagou.run()
